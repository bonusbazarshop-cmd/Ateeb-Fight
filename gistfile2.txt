class Game {
    constructor() {
        this.canvas = document.getElementById('gameCanvas');
        this.ctx = this.canvas.getContext('2d');
        this.width = this.canvas.width;
        this.height = this.canvas.height;
        this.STATE = { MENU: 'MENU', LEVEL_SELECT: 'LEVEL_SELECT', PLAYING: 'PLAYING', PAUSED: 'PAUSED', GAME_OVER: 'GAME_OVER', LEVEL_COMPLETE: 'LEVEL_COMPLETE', LEADERBOARD: 'LEADERBOARD', TWO_PLAYER: 'TWO_PLAYER' };
        this.currentState = this.STATE.MENU;
        this.currentLevel = 1;
        this.gameMode = null;
        this.isPaused = false;
        this.players = [];
        this.enemies = [];
        this.bullets = [];
        this.powerups = [];
        this.particles = [];
        this.soundManager = new SoundManager();
        this.scoreManager = new ScoreManager();
        this.levelManager = new LevelManager();
        this.hudManager = new HUDManager(this);
        this.menuManager = new MenuManager(this);
        this.screenManager = new ScreenManager(this);
        this.levelStartTime = 0;
        this.elapsedTime = 0;
        this.frameCount = 0;
        this.fps = 60;
        this.init();
    }
    init() {
        this.setupInputHandlers();
        this.menuManager.setupMenuListeners();
        this.screenManager.setupScreenListeners();
        this.startGameLoop();
    }
    setupInputHandlers() {
        this.keysPressed = {};
        window.addEventListener('keydown', (e) => {
            this.keysPressed[e.key.toLowerCase()] = true;
            this.handleKeyDown(e);
        });
        window.addEventListener('keyup', (e) => {
            this.keysPressed[e.key.toLowerCase()] = false;
        });
    }
    handleKeyDown(e) {
        if (e.key === 'Escape' && this.currentState === this.STATE.PLAYING) this.togglePause();
    }
    togglePause() {
        if (this.currentState === this.STATE.PLAYING) this.isPaused = !this.isPaused;
    }
    startLevel(levelNumber, mode = 'single') {
        this.currentLevel = levelNumber;
        this.gameMode = mode;
        this.currentState = this.STATE.PLAYING;
        this.isPaused = false;
        this.levelStartTime = Date.now();
        this.frameCount = 0;
        this.elapsedTime = 0;
        this.players = [];
        this.enemies = [];
        this.bullets = [];
        this.powerups = [];
        this.particles = [];
        if (mode === 'single') {
            this.players.push(new Player(this.width / 2, this.height / 2, 1, this));
        } else if (mode === 'twoPlayer') {
            this.players.push(new Player(this.width / 4, this.height / 2, 1, this));
            this.players.push(new Player((3 * this.width) / 4, this.height / 2, 2, this));
        }
        const levelConfig = this.levelManager.getLevelConfig(levelNumber);
        this.levelManager.currentLevel = levelNumber;
        this.levelManager.spawnWave(0, levelConfig);
        this.scoreManager.resetLevelScore();
        this.menuManager.hideAllScreens();
    }
    endLevel(success = false) {
        this.currentState = this.STATE.GAME_OVER;
        const stats = this.scoreManager.getLevelStats();
        this.screenManager.showGameOverScreen(stats);
    }
    completeLevel() {
        this.currentState = this.STATE.LEVEL_COMPLETE;
        const stats = this.scoreManager.getLevelStats();
        this.screenManager.showLevelCompleteScreen(stats);
    }
    update() {
        if (this.currentState !== this.STATE.PLAYING || this.isPaused) return;
        this.frameCount++;
        this.elapsedTime = (Date.now() - this.levelStartTime) / 1000;
        this.players.forEach((player) => player.update(this.keysPressed));
        this.enemies.forEach((enemy, index) => {
            enemy.update(this.players);
            if (enemy.health <= 0) {
                this.enemies.splice(index, 1);
                this.scoreManager.addKill();
                this.soundManager.playEnemyDeathSound();
                this.createExplosion(enemy.x, enemy.y);
            }
        });
        this.bullets.forEach((bullet, index) => {
            bullet.update();
            if (bullet.isOutOfBounds(this.width, this.height)) this.bullets.splice(index, 1);
        });
        this.powerups.forEach((powerup, index) => {
            powerup.update();
            if (powerup.isOutOfBounds(this.width, this.height)) this.powerups.splice(index, 1);
        });
        this.particles.forEach((particle, index) => {
            particle.update();
            if (particle.isDead()) this.particles.splice(index, 1);
        });
        this.checkCollisions();
        this.levelManager.update(this.enemies.length);
        if (this.levelManager.isLevelComplete()) this.completeLevel();
        this.players.forEach((player) => {
            if (player.health <= 0) this.endLevel(false);
        });
    }
    checkCollisions() {
        this.bullets.forEach((bullet, bulletIndex) => {
            this.enemies.forEach((enemy) => {
                if (CollisionDetection.checkAABB(bullet, enemy)) {
                    enemy.takeDamage(bullet.damage);
                    this.bullets.splice(bulletIndex, 1);
                    this.soundManager.playHitSound();
                    this.createExplosion(bullet.x, bullet.y);
                    this.scoreManager.addScore(100);
                }
            });
        });
        this.enemies.forEach((enemy) => {
            this.players.forEach((player) => {
                if (CollisionDetection.checkAABB(enemy, player)) {
                    player.takeDamage(10);
                    enemy.takeDamage(5);
                    this.soundManager.playPlayerHitSound();
                }
            });
        });
        this.powerups.forEach((powerup, index) => {
            this.players.forEach((player) => {
                if (CollisionDetection.checkAABB(powerup, player)) {
                    player.applyPowerup(powerup);
                    this.powerups.splice(index, 1);
                    this.soundManager.playPowerupSound();
                }
            });
        });
    }
    createExplosion(x, y, particleCount = 30) {
        for (let i = 0; i < particleCount; i++) {
            const angle = (Math.PI * 2 * i) / particleCount;
            const velocity = 3 + Math.random() * 3;
            this.particles.push(new Particle(x, y, angle, velocity));
        }
    }
    draw() {
        this.ctx.fillStyle = 'rgba(10, 0, 21, 0.3)';
        this.ctx.fillRect(0, 0, this.width, this.height);
        this.drawStarfield();
        if (this.currentState === this.STATE.PLAYING) {
            this.enemies.forEach((enemy) => enemy.draw(this.ctx));
            this.players.forEach((player) => player.draw(this.ctx));
            this.bullets.forEach((bullet) => bullet.draw(this.ctx));
            this.powerups.forEach((powerup) => powerup.draw(this.ctx));
            this.particles.forEach((particle) => particle.draw(this.ctx));
            this.hudManager.draw(this.ctx);
            if (this.isPaused) this.drawPauseOverlay();
        }
    }
    drawStarfield() {
        this.ctx.fillStyle = 'rgba(255, 255, 255, 0.3)';
        for (let i = 0; i < 50; i++) {
            const x = (this.frameCount * 0.5 + i * 100) % this.width;
            const y = (i * 15.36) % this.height;
            this.ctx.beginPath();
            this.ctx.arc(x, y, 1, 0, Math.PI * 2);
            this.ctx.fill();
        }
    }
    drawPauseOverlay() {
        this.ctx.fillStyle = 'rgba(0, 0, 0, 0.7)';
        this.ctx.fillRect(0, 0, this.width, this.height);
        this.ctx.fillStyle = '#ffee00';
        this.ctx.font = 'bold 3rem Fredoka One';
        this.ctx.textAlign = 'center';
        this.ctx.fillText('PAUSED', this.width / 2, this.height / 2);
        this.ctx.font = 'bold 1.2rem Nunito';
        this.ctx.fillText('Press ESC to resume', this.width / 2, this.height / 2 + 60);
    }
    startGameLoop() {
        const gameLoop = () => {
            this.update();
            this.draw();
            requestAnimationFrame(gameLoop);
        };
        gameLoop();
    }
}
let game;
window.addEventListener('DOMContentLoaded', () => { game = new Game(); });
